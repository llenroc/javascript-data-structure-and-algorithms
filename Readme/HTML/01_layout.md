### Appropriate tag to divide document into logical section

<header>
<nav>
<main>
<article>
<section>
<aside>
<footer>


```
<!DOCTYPE html>
<html>
  <head>
    <title> My page Title</title>
  </head>
  <body>
    <header id = "page-header"> 
      <h1>Site title</h1>
      <form> Search </form>
      <nav class="site-nav">
        <ul>Site navigation</ul>
      </nav>
    </header>
    <main>
      <section id = "main-content">
        <article>
          <h2>Article title</h1>
          <p>Summary</p>
        </article>
        <article>
          <h2>Article title</h1>
          <p>Summary</p>
        </article>
        <article>
          <h2>Article title</h1>
          <p>Summary</p>
        </article>
      </section>
      <aside class="sidebar">
        <section>
          <h2>Blogroll...</h2>
        </section>
        <section>
          <h2>Photos...</h2>
        </section>
      </aside>
    </main>
    <footer id="page-footer">
      <h2>Footer</h2>
      <nav class="site-footer-nav">
        <ul>Site navigation</ul>
      </nav>
    </footer>
  </body>
</html>
```
